# 예제 7: 컴파일 타임 메타프로그래밍

> `compile_eval`, `static_if`, `dim` 을 결합해
> 런타임 비용 없이 복잡한 계산을 컴파일 타임에 완료한다.

---

## 개요

- `compile_eval` : 함수를 `constexpr` 로 변환 → 컴파일 타임 실행
- `static_if` : 컴파일 타임 조건 분기 (이진 배포 가능)
- `dim` + `compile_eval` : 단위 변환 상수 컴파일 타임 계산
- `-D` 플래그로 빌드 프로파일 선택

---

## 코드

```dri
module meta;

# ── 1. 컴파일 타임 수학 함수 ─────────────────────────────────

compile_eval int fibonacci(int n) {
    if (n <= 1) { return n; }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

compile_eval int factorial(int n) {
    if (n <= 1) { return 1; }
    return n * factorial(n - 1);
}

compile_eval boolean is_prime(int n) {
    if (n < 2) { return false; }
    for (i in 2..n) {
        if (n % i == 0) { return false; }
    }
    return true;
}

# 컴파일 타임에 계산 — 런타임에는 상수로 사용
int fib20  = fibonacci(20);   # 컴파일 타임: 6765
int fac12  = factorial(12);   # 컴파일 타임: 479001600
boolean p7 = is_prime(7);     # 컴파일 타임: true

print(`fib(20) = ${fib20}`);
print(`12! = ${fac12}`);
print(`is_prime(7) = ${p7}`);

# ── 2. 컴파일 타임 룩업 테이블 ──────────────────────────────

compile_eval int[] precompute_fibonacci_table(int n) {
    int[] table = lst.fill(n, 0);
    table[0] = 0;
    if (n > 1) { table[1] = 1; }
    for (i in 2..n) {
        table[i] = table[i-1] + table[i-2];
    }
    return table;
}

# 컴파일 타임에 생성된 피보나치 테이블
int[] FIB_TABLE = precompute_fibonacci_table(30);
print(`fib(25) = ${FIB_TABLE[25]}`);  # O(1) 룩업

# ── 3. 물리 단위 컴파일 타임 상수 ───────────────────────────

dim Meter   = "m";
dim Foot    = "ft";
dim Celsius = "°C";
dim Kelvin  = "K";

# 단위 변환 상수 (컴파일 타임 계산)
compile_eval double feet_to_meters(double feet) {
    return feet * 0.3048;
}

compile_eval double celsius_to_kelvin(double c) {
    return c + 273.15;
}

compile_eval double meters_to_feet(double meters) {
    return meters / 0.3048;
}

double EMPIRE_STATE_METERS = feet_to_meters(1454.0);  # 컴파일 타임
double BOILING_KELVIN      = celsius_to_kelvin(100.0); # 컴파일 타임
double MARATHON_FEET       = meters_to_feet(42195.0);  # 컴파일 타임

print(`엠파이어 스테이트: ${EMPIRE_STATE_METERS} m`);
print(`끓는점: ${BOILING_KELVIN} K`);
print(`마라톤: ${MARATHON_FEET} ft`);

# ── 4. static_if — 빌드 프로파일별 코드 ──────────────────────

static_if (DEBUG) {
    # 디버그 빌드: 상세 로깅, 경계 검사
    print("[DEBUG] 디버그 모드로 빌드됨");

    void checked_get(int[] arr, int idx) {
        if (idx < 0 or idx >= lst.length(arr)) {
            throw `인덱스 범위 초과: ${idx} / ${lst.length(arr)}`;
        }
    }

} else {
    static_if (RELEASE) {
        # 릴리스 빌드: 최적화, 로깅 제거
        print("[RELEASE] 릴리스 모드");

    } else {
        # 기본 빌드
        print("[DEFAULT] 기본 빌드");
    }
}

# 플랫폼별 분기
static_if (WINDOWS) {
    String PATH_SEP = "\\";
    String CONFIG_DIR = "%APPDATA%\\myapp";
} else {
    static_if (MACOS) {
        String PATH_SEP = "/";
        String CONFIG_DIR = "~/Library/Application Support/myapp";
    } else {
        String PATH_SEP = "/";
        String CONFIG_DIR = "~/.config/myapp";
    }
}

# ── 5. 타입별 컴파일 타임 특수화 ────────────────────────────

# T에 따라 다른 알고리즘 선택
compile_eval boolean is_floating_point_type(String type_name) {
    return type_name == "double" or type_name == "float";
}

void compute<T>(T a, T b) {
    static_if (EXACT_ARITHMETIC) {
        # 정수 정밀 계산
        print(`정수 결과: ${a + b}`);
    } else {
        # 부동소수점 계산
        print(`실수 결과: ${a + b}`);
    }
}

# ── 6. 컴파일 타임 소수 체 ──────────────────────────────────

compile_eval int[] sieve_of_eratosthenes(int limit) {
    boolean[] is_prime_arr = lst.fill(limit + 1, true);
    is_prime_arr[0] = false;
    if (limit > 0) { is_prime_arr[1] = false; }

    for (i in 2..limit) {
        if (is_prime_arr[i]) {
            int j = i * i;
            while (j <= limit) {
                is_prime_arr[j] = false;
                j += i;
            }
        }
    }

    int[] primes = [];
    for (i in 2..limit) {
        if (is_prime_arr[i]) {
            lst.push(primes, i);
        }
    }
    return primes;
}

# 컴파일 타임에 100 이하 소수 모두 계산
int[] PRIMES_100 = sieve_of_eratosthenes(100);

print(`100 이하 소수 개수: ${lst.length(PRIMES_100)}`);
print("처음 10개:");
for (p of lst.take(PRIMES_100, 10)) {
    print(p);
}

# ── 7. 행렬 크기 컴파일 타임 최적화 ─────────────────────────

compile_eval int next_power_of_2(int n) {
    int p = 1;
    while (p < n) { p = p * 2; }
    return p;
}

int OPTIMAL_SIZE = next_power_of_2(500);   # 컴파일 타임: 512
print(`최적 행렬 크기 (500→): ${OPTIMAL_SIZE}`);
```

---

## 컴파일 타임 vs 런타임 비교

| 계산 | 없이 | `compile_eval` |
|------|------|----------------|
| `fibonacci(20)` | 런타임 재귀 6765회 | 컴파일 타임 1회 → 상수 |
| `sieve(100)` | 매 실행마다 계산 | 컴파일 타임 → 배열 상수 |
| `static_if (DEBUG)` | 런타임 `if` | 컴파일 타임 제거/포함 |

## 빌드 명령

```bash
# 디버그 빌드
dri main.dri --exe app -DDEBUG

# 릴리스 빌드

# Windows 빌드
dri main.dri --exe app -DWINDOWS

# 여러 플래그 동시
dri main.dri --exe app --release -DRELEASE -DWINDOWS
```

---

## 학습 포인트

- `compile_eval` = `constexpr` (C++) 자동 변환
- `static_if (FLAG)` = 이진 파일에서 조건부 코드 완전 제거
- 룩업 테이블 패턴: O(n) 전처리 → O(1) 런타임 접근
- `dim` + `compile_eval` = 단위 변환 오류 컴파일 타임 차단
